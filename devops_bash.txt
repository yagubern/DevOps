48. Написать скрипт
#!/bin/bash
file=$HOME/file.txt
echo $1
function check_writable () {
 if [ -w "$file" ]
    then
    echo $1 >> $file
    return 0
    else return 1
 fi
}
check_writable $1 $file
exit $?
